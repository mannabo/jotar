// Firestore Security Rules for Production Mode
// Copy this to Firebase Console > Firestore Database > Rules

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Activities collection - Only authenticated users can read/write
    match /activities/{activityId} {
      allow read, write: if request.auth != null;
    }
    
    // Messages collection - Only authenticated users can read/write
    match /messages/{messageId} {
      allow read, write: if request.auth != null;
    }
    
    // Users collection (if needed later)
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    
    // Admin settings (restrict to specific admin emails)
    match /admin_settings/{document} {
      allow read, write: if request.auth != null && 
        request.auth.token.email in [
          'admin@jotar.org',
          'jotar.johor@gmail.com'
        ];
    }
    
    // Public data (if needed - like public announcements)
    match /public/{document} {
      allow read: if true;
      allow write: if request.auth != null;
    }
  }
}
